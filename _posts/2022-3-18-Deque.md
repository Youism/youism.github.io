package deque;

import java.util.Iterator;

public class ArrayDeque<T> implements Deque<T>, Iterable<T> {

    private int size;
    private T[] items;
    private int nextFirst;
    private int nextLast;

    public ArrayDeque() {
        this.items = (T[]) new Object[8];
        size = 0;
        nextFirst = 0;
        nextLast = 0;
    }

    /**
     * add to nextFirst
     */
    @Override
    public void addFirst(T item) {
        if (size() == this.items.length) {
            resize(size() * 2);
        }
        this.items[nextFirst] = item;
        if (nextFirst == nextLast && nextFirst == 0) {
            this.nextLast++;
        }
        this.nextFirst = (nextFirst - 1 + items.length) % items.length;
        this.size++;
    }

    private void resize(int i) {
        var arr = (T[]) new Object[i];
        for (int j = 0; j < size(); j++) {
            arr[j] = get(j);
        }
        this.nextFirst = i - 1;
        this.nextLast = size();
        this.items = arr;
    }

    @Override
    public void addLast(T item) {
        if (size() == this.items.length) {
            resize(size() * 2);
        }
        this.items[nextLast] = item;
        if (nextFirst == nextLast && nextFirst == 0) {
            this.nextFirst = items.length - 1;
        }
        this.nextLast = (nextLast + 1) % items.length;
        this.size++;
    }

    @Override
    public int size() {
        return this.size;
    }

    @Override
    public void printDeque() {
        var index = (this.nextFirst + 1) % items.length;
        while (index != nextLast) {
            System.out.println(items[index] + " ");
            index = (index + 1) % items.length;
        }
        System.out.println();
    }

    @Override
    public T removeFirst() {
        if (isEmpty()) return null;
        var index = (nextFirst + 1) % items.length;
        var result = items[index];
        items[index] = null;
        nextFirst = index;
        this.size--;
        if (size() < 0.25 * items.length && size() > 8) resize(items.length / 2);
        return result;
    }

    @Override
    public T removeLast() {
        if (isEmpty()) return null;
        var index = (nextLast - 1 + items.length) % items.length;
        var result = items[index];
        items[index] = null;
        nextLast = index;
        this.size--;
        if (size() < 0.25 * items.length && size() > 8) resize(items.length / 2);
        return result;
    }

    @Override
    public T get(int index) {
        if (index >= size()) return null;
        return items[(nextFirst + index + 1) % items.length];
    }

    @Override
    public Iterator<T> iterator() {
        return new ArrayDequeIterator();
    }

    private class ArrayDequeIterator implements Iterator<T> {

        private int pos;

        ArrayDequeIterator() {
            pos = (nextFirst + 1 + items.length) % items.length;
        }

        @Override
        public boolean hasNext() {
            return pos != nextLast;
        }

        @Override
        public T next() {
            var result = items[pos];
            pos++;
            return result;
        }
    }

    @Override
    public boolean equals(Object o) {
        if (o == this) return false;
        if (o == null) return false;
        if (!(o instanceof ArrayDeque)) return false;
        ArrayDeque<T> other = (ArrayDeque<T>) o;
        if (other.size() != this.size()) return false;
        var index = (this.nextFirst + 1) % items.length;
        var index1 = (other.nextFirst + 1) % other.items.length;
        while (index != nextLast) {
            if (items[index] != other.items[index1]) return false;
            index = (index + 1) % items.length;
            index1 = (index1 + 1) % other.items.length;
        }
        return true;
    }
}

package deque;

import javax.print.attribute.standard.MediaSize;
import java.util.Iterator;

public class LinkedListDeque<T> implements Deque<T>, Iterable<T> {
    private int size;
    Node sentinel;

    @Override
    public Iterator<T> iterator() {
        return new LinkedListDequeIterator();
    }

    private class LinkedListDequeIterator implements Iterator<T> {
        private Node pos;

        public LinkedListDequeIterator() {
            pos = sentinel.next;
        }

        @Override
        public boolean hasNext() {
            return pos != sentinel;
        }

        @Override
        public T next() {
            var resultItem = pos.item;
            pos = pos.next;
            return resultItem;
        }
    }

    private class Node {
        public T item;
        public Node next;
        public Node prev;

        Node() {
            item = null;
            next = this;
            prev = this;
        }

        public Node(T item, Node next, Node prev) {
            this.item = item;
            this.prev = prev;
            this.next = next;
        }
    }

    LinkedListDeque() {
        this.size = 0;
        sentinel = new Node();
    }

    @Override
    public void addFirst(T item) {
        this.size++;
        Node p = new Node(item, sentinel, sentinel.next);
        sentinel.next.prev = p;
        sentinel.next = p;
    }

    @Override
    public void addLast(T item) {
        this.size++;
        var p = new Node(item, sentinel, sentinel.prev);
        sentinel.prev.next = p;
        sentinel.prev = p;
    }

    @Override
    public int size() {
        return this.size;
    }

    @Override
    public void printDeque() {
        var tmp = sentinel.next;
        while (tmp != sentinel) {
            System.out.print(tmp.item + " ");
            tmp = tmp.next;
        }
        System.out.println();
    }

    @Override
    public T removeFirst() {
        if (isEmpty()) return null;
        sentinel.next.next.prev = sentinel;
        var result = sentinel.next.item;
        sentinel.next = sentinel.next.next;
        this.size--;
        return result;
    }

    @Override
    public T removeLast() {
        if (isEmpty()) return null;
        sentinel.prev.prev.next = sentinel;
        var result = sentinel.prev.item;
        sentinel.prev = sentinel.prev.prev;
        this.size--;
        return result;
    }

    @Override
    public T get(int index) {
        if (index >= size()) return null;
        var p = sentinel.next;
        while (index > 0) {
            p = p.next;
            index--;
        }
        return p.item;
    }

    public T getRecursive(int index) {
        if (index >= size()) return null;
        var p = sentinel;
        return helper(p, index);
    }

    private T helper(Node s, int index) {
        if (index == 0) return s.next.item;
        return helper(s.next, index - 1);
    }
    @Override
    public boolean equals(Object o){
        if(this == o) return true;
        if(o == null) return false;
        if(! (o instanceof LinkedListDeque)){
           return false;
        }
        LinkedListDeque<T> other = (LinkedListDeque<T>) o;
        if(other.size() != this.size()) return false;
        var s1 = this.sentinel.next;
        var s2 = other.sentinel.next;
        while (s1 != sentinel){
            if(s1.item != s2.item) return false;
            s1 = s1.next;
            s2 = s2.next;
        }
        return true;
    }
}
